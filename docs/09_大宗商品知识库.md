# 09_大宗商品知识库设计文档

## 概述
大宗商品知识库是期货AI交易系统的扩展模块，旨在从贵金属（黄金、白银）扩展到全品类大宗商品分析，构建完整的商品市场认知体系，帮助用户识别不同品种的投资机会、理解品种间关联关系、挖掘潜力标的。

## 设计目标

### 核心功能
1. **全品种覆盖**：能源、金属、农产品、软商品等主要大宗商品
2. **关联网络分析**：构建品种间相关性、领先滞后关系、因果关系网络
3. **潜力标的挖掘**：基于多维度的品种评分和投资机会识别
4. **知识图谱构建**：商品属性、产业链、宏观经济关系的知识图谱
5. **投资策略推荐**：基于市场环境的品种轮动和策略建议

```python
# knowledge_base/correlation.py
class CommodityCorrelationAnalyzer:
    """商品相关性分析"""
    
    def __init__(self):
        self.correlation_methods = {
            'pearson': self._pearson_correlation,
            'spearman': self._spearman_correlation,
            'dynamic': self._dynamic_correlation,
            'rolling': self._rolling_correlation
        }
    
    def analyze_correlation_network(self, commodities, period='1y'):
        """分析商品关联网络"""
        # 1. 获取价格数据
        price_data = self._fetch_price_data(commodities, period)
        
        # 2. 计算相关性矩阵
        corr_matrix = self._calculate_correlation_matrix(price_data)
        
        # 3. 识别强相关对
        strong_pairs = self._identify_strong_correlations(corr_matrix)
        
        # 4. 构建关联网络
        network = self._build_correlation_network(strong_pairs)
        
        return {
            'correlation_matrix': corr_matrix,
            'strong_pairs': strong_pairs,
            'network': network,
            'centrality': self._calculate_centrality(network)
        }
    
    def _identify_strong_correlations(self, corr_matrix, threshold=0.7):
        """识别强相关商品对"""
        strong_pairs = []
        n = len(corr_matrix)
        
        for i in range(n):
            for j in range(i+1, n):
                corr = corr_matrix[i][j]
                if abs(corr) >= threshold:
                    strong_pairs.append({
                        'commodity1': self.commodities[i],
                        'commodity2': self.commodities[j],
                        'correlation': corr,
                        'type': 'positive' if corr > 0 else 'negative'
                    })
        
        return sorted(strong_pairs, key=lambda x: abs(x['correlation']), reverse=True)
```

#### 1.2 领先滞后关系分析
```python
def analyze_lead_lag_relationships(self, commodity_pairs, max_lag=20):
    """分析领先滞后关系"""
    results = []
    
    for pair in commodity_pairs:
        commodity_a, commodity_b = pair
        
        # 获取价格序列
        series_a = self._get_price_series(commodity_a)
        series_b = self._get_price_series(commodity_b)
        
        # 计算交叉相关性
        cross_corr = self._cross_correlation(series_a, series_b, max_lag)
        
        # 找到最大相关性的滞后
        max_corr = max(cross_corr, key=abs)
        lag_at_max = cross_corr.index(max_corr) - max_lag
        
        results.append({
            'leading': commodity_a if lag_at_max < 0 else commodity_b,
            'lagging': commodity_b if lag_at_max < 0 else commodity_a,
            'lag_periods': abs(lag_at_max),
            'correlation': max_corr,
            'confidence': self._calculate_confidence(cross_corr)
        })
    
    return results
```

```python
# knowledge_base/scoring.py
class CommodityScoringModel:
    """商品多维评分模型"""
    
    def __init__(self):
        self.dimensions = {
            'momentum': self._score_momentum,
            'value': self._score_value,
            'quality': self._score_quality,
            'sentiment': self._score_sentiment,
            'macro': self._score_macro
        }
        
        self.weights = {
            'momentum': 0.25,
            'value': 0.20,
            'quality': 0.20,
            'sentiment': 0.15,
            'macro': 0.20
        }
    
    def score_commodity(self, commodity, lookback_period='3m'):
        """对商品进行综合评分"""
        scores = {}
        
        # 计算各维度分数
        for dim_name, scoring_func in self.dimensions.items():
            score = scoring_func(commodity, lookback_period)
            scores[dim_name] = {
                'score': score,
                'weight': self.weights[dim_name],
                'weighted_score': score * self.weights[dim_name]
            }
        
        # 计算综合分数
        total_score = sum(scores[dim]['weighted_score'] for dim in scores)
        
        # 生成投资建议
        recommendation = self._generate_recommendation(total_score, scores)
        
        return {
            'commodity': commodity,
            'total_score': total_score,
            'dimension_scores': scores,
            'recommendation': recommendation,
            'timestamp': datetime.now()
        }
    
    def _score_momentum(self, commodity, lookback_period):
        """动量评分"""
        price_data = self._get_price_data(commodity, lookback_period)
        
        # 计算动量指标
        returns = price_data.pct_change()
        momentum_1m = returns[-20:].mean() * 20  # 月动量
        momentum_3m = returns[-60:].mean() * 60  # 季动量
        
        # 趋势强度
        trend_strength = self._calculate_trend_strength(price_data)
        
        # 动量分数 (0-100)
        score = 50 + (momentum_1m * 1000) + (momentum_3m * 500) + (trend_strength * 20)
        return max(0, min(100, score))
    
    def _score_value(self, commodity, lookback_period):
        """价值评分"""
        # 获取基本面数据
        fundamentals = self._get_fundamentals(commodity)
        
        # 计算估值指标
        price_to_cost = self._calculate_price_to_cost(fundamentals)
        inventory_level = self._normalize_inventory(fundamentals['inventory'])
        backwardation = self._calculate_backwardation(commodity)
        
        # 价值分数
        score = 50
        score += (1 / price_to_cost - 1) * 20  # 价格/成本比越低越好
        score += inventory_level * 15  # 库存水平适中为好
        score += backwardation * 15  # 现货升水为正价值
        
        return max(0, min(100, score))
```

#### 2.2 投资机会识别
```python
def identify_investment_opportunities(self, top_n=10):
    """识别投资机会"""
    all_commodities = self._get_all_commodities()
    scored_commodities = []
    
    # 对所有商品评分
    for commodity in all_commodities:
        score_result = self.score_commodity(commodity)
        scored_commodities.append(score_result)
    
    # 按总分排序
    scored_commodities.sort(key=lambda x: x['total_score'], reverse=True)
    
    # 分类机会
    opportunities = {
        'momentum_plays': [],  # 动量交易机会
        'value_plays': [],     # 价值投资机会
        'contrarian_plays': [], # 逆向投资机会
        'hedge_opportunities': [] # 对冲机会
    }
    
    for i, commodity in enumerate(scored_commodities[:top_n]):
        scores = commodity['dimension_scores']
        
        # 动量机会 (动量分高，价值分中等)
        if scores['momentum']['score'] > 70 and scores['value']['score'] > 40:
            opportunities['momentum_plays'].append(commodity)
        
        # 价值机会 (价值分高，动量分中等)
        elif scores['value']['score'] > 70 and scores['momentum']['score'] > 40:
            opportunities['value_plays'].append(commodity)
        
        # 逆向机会 (动量分低但价值分高)
        elif scores['momentum']['score'] < 30 and scores['value']['score'] > 60:
            opportunities['contrarian_plays'].append(commodity)
        
        # 对冲机会 (与其他商品负相关性强)
        hedge_pairs = self._find_hedge_pairs(commodity['commodity'])
        if hedge_pairs:
            opportunities['hedge_opportunities'].append({
                'commodity': commodity,
                'hedge_pairs': hedge_pairs
            })
    
    return opportunities
```

```python
# knowledge_base/rotation.py
class CommodityRotationStrategy:
    """商品轮动策略"""
    
    def __init__(self):
        self.economic_cycles = {
            'early_recession': self._early_recession_allocation,
            'late_recession': self._late_recession_allocation,
            'early_recovery': self._early_recovery_allocation,
            'late_recovery': self._late_recovery_allocation,
            'early_expansion': self._early_expansion_allocation,
            'late_expansion': self._late_expansion_allocation
        }
    
    def determine_economic_cycle(self, macro_indicators):
        """确定当前经济周期阶段"""
        # 基于宏观指标判断周期阶段
        gdp_growth = macro_indicators.get('gdp_growth', 0)
        inflation = macro_indicators.get('inflation', 0)
        unemployment = macro_indicators.get('unemployment', 0)
        yield_curve = macro_indicators.get('yield_curve', 0)
        
        # 周期判断逻辑
        if gdp_growth < 0 and inflation < 2:
            return 'early_recession'
        elif gdp_growth < 0 and inflation > 2:
            return 'late_recession'
        elif gdp_growth > 2 and unemployment > 6:
            return 'early_recovery'
        elif gdp_growth > 3 and unemployment < 6:
            return 'late_recovery'
        elif gdp_growth > 4 and inflation < 3:
            return 'early_expansion'
        elif gdp_growth > 3 and inflation > 3:
            return 'late_expansion'
        else:
            return 'uncertain'
    
    def get_optimal_allocation(self, cycle_stage):
        """获取最优商品配置"""
        allocation_func = self.economic_cycles.get(cycle_stage)
        if allocation_func:
            return allocation_func()
        else:
            return self._neutral_allocation()
    
    def _early_recession_allocation(self):
        """早期衰退阶段配置"""
```
